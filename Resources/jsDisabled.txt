function EmptyDoc(pVariable)
{
	var result = false;

	if(pVariable === undefined)
	{
		result= true;
	}else if(pVariable === null)
	{
		result= true;
	}
	else if(pVariable === "")
	{
		result= true;
	}
	else if(pVariable === NaN)
	{
		result= true;
	}
	else if(pVariable.length <= 0)
	{
		result= true;
	}

	return result;
}
